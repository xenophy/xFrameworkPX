  149.     public function save($pFilename = null) {

  151.         // check mbstring.func_overload
  152.         if (ini_get('mbstring.func_overload') != 0) {

  161.  
  162.         // Initialise workbook writer

  164.                     $this->_str_total, $this->_str_unique, $this->_str_table, $this->_colors, $this->_parser, $this->_tempDir);
  165.

  174.                                        $this->_parser, $this->_tempDir,

  177.         }
  178.

  191.         // initialize OLE file
  192.         $workbookStreamName = ($this->_BIFF_version == 0x0600) ? 'Workbook' : 'Book';
  193.         $OLE = new PHPExcel_Shared_OLE_PPS_File(PHPExcel_Shared_OLE::Asc2Ucs($workbookStreamName));

  200.         // Write the worksheet streams before the global workbook stream,

  202.         $worksheetSizes = array();

  204.             $this->_writerWorksheets[$i]->close();

  206.         }
  207.

  212.         for ($i = 0; $i < $countSheets; ++$i) {

  214.                 $OLE->append($tmp);
  215.             }
  216.         }
  217.  
  218.         $root = new PHPExcel_Shared_OLE_PPS_Root(time(), time(), array($OLE));
  219.         if ($this->_tempDir != '') {
  220.             $root->setTempDir($this->_tempDir);
  221.         }
  222.  
  223.         // save the OLE file
